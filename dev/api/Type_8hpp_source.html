341 // Real counterpart of a dtype: ComplexDouble -> Double, ComplexFloat -> Float,
342 // anything else unchanged. Replaces the "dtype <= 2 ? dtype + 2 : dtype" idiom
343 // (without depending on the enum layout).
344 static constexpr unsigned int to_real(unsigned int type_id) {
345 check_type(type_id);
346 if (type_id == ComplexDouble) return Double;
347 if (type_id == ComplexFloat) return Float;
348 return type_id;
349 }
350
351 // Complex counterpart of a dtype: Double -> ComplexDouble, Float -> ComplexFloat,
352 // complex types unchanged, Void unchanged, integral/bool -> ComplexDouble.
353 static constexpr unsigned int to_complex(unsigned int type_id) {
354 check_type(type_id);
355 if (is_complex(type_id)) return type_id;
356 if (type_id == Double) return ComplexDouble;
357 if (type_id == Float) return ComplexFloat;
358 if (type_id == Void) return Void;
359 return ComplexDouble;
360 }

362 // Find a common type for typeL and typeR
363 static constexpr unsigned int type_promote(unsigned int typeL, unsigned int typeR) {
364 if (typeL == Void || typeR == Void) return Void;
365 // Mixed complex/real: promote the real counterparts, then re-complexify.
366 // Fixes ComplexFloat + Double -> ComplexDouble (previously ComplexFloat,
367 // discarding precision, because the enum interleaves complexness and
368 // precision and promotion picked the lower index).
369 if (is_complex(typeL) != is_complex(typeR)) {
370 return to_complex(type_promote(to_real(typeL), to_real(typeR)));
371 }
372 if (typeL < typeR) {
373 if (!is_unsigned(typeR) && is_unsigned(typeL)) {
374 return typeL - 1;
375 } else {
376 return typeL;
377 }
378 } else {
379 if (!is_unsigned(typeL) && is_unsigned(typeR)) {
380 return typeR - 1;
381 } else {
382 return typeR;
383 }
384 }
385 }
