498 struct exclude_first;

500 template <typename First, typename... Rest>
501 struct exclude_first<std::variant<First, Rest...>> {
502 using type = std::variant<Rest...>;
503 };

507 // std::variant of pointers to Type_list, without void ....
508 using pointer_types =
509 make_variant_from_transform_t<typename internal::exclude_first<Type_list>::type,
510 std::add_pointer>;
511
512 // convert this->_impl->_storage._impl->Mem to a typed variant of pointers, excluding void*
513 pointer_types ptr() const;
514
515 // convert this->_impl->_strorage->Mem to the given pointer type.
516 // Throws an exception if T does not match this->dtype
517 template <typename T>
518 T *ptr_as(bool check_type = true) const {
519 if (check_type) {
520 cytnx_error_msg(this->dtype() != Type_class::cy_typeid_v<std::remove_cv_t<T>>,
521 "[ERROR] Attempt to convert dtype %d (%s) to pointer of type %s",
522 this->dtype(), Type_class::getname(this->dtype()).c_str(),
523 Type_class::getname(Type_class::cy_typeid_v<std::remove_cv_t<T>>).c_str());
524 }
525 return reinterpret_cast<T *>(this->_impl->_storage._impl->data());
526 }
