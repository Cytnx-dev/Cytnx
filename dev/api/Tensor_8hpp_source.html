496 struct internal {
497 template <typename Variant>
498 struct exclude_first;

500 template <typename First, typename... Rest>
501 struct exclude_first<std::variant<First, Rest...>> {
502 using type = std::variant<Rest...>;
503 };

507 // std::variant of pointers to Type_list, without void ....
508 using pointer_types =
509 make_variant_from_transform_t<typename internal::exclude_first<Type_list>::type,
510 std::add_pointer>;
511
512 // convert this->_impl->_storage._impl->Mem to a typed variant of pointers, excluding void*
513 pointer_types ptr() const;
514
515 // Convert storage to the logical Cytnx element pointer type.
516 //
517 // Use ptr_as<T>() for host code and raw memory copies. For complex tensors, T is
518 // std::complex<...>. Use gpu_ptr_as<T>() for CUDA kernel code; for complex tensors, T is
519 // cuda::std::complex<...>. CUDA library APIs that require cuComplex ABI pointers should cast
520 // explicitly at the library-call boundary.
521 template <typename T>
522 T *ptr_as() const {
523 cytnx_error_msg(this->dtype() != Type_class::cy_typeid_v<std::remove_cv_t<T>>,
524 "[ERROR] Attempt to convert dtype %d (%s) to pointer of type %s",
525 this->dtype(), Type_class::getname(this->dtype()).c_str(),
526 Type_class::getname(Type_class::cy_typeid_v<std::remove_cv_t<T>>).c_str());
527 return static_cast<T *>(this->_impl->_storage._impl->data());
528 }
529
530 #ifdef UNI_GPU
531 // std::variant of pointers to Type_list_gpu, without void ....
532 using gpu_pointer_types =
533 make_variant_from_transform_t<typename internal::exclude_first<Type_list_gpu>::type,
534 std::add_pointer>;
535
536 // convert this->_impl->_storage->Mem to a typed variant of pointers, excluding void*
537 gpu_pointer_types gpu_ptr() const;
538
539 // Convert storage to the CUDA kernel element pointer type.
540 //
541 // This is for CUDA kernel interfaces. Generic host code should use ptr_as<T>() so the logical
542 // Cytnx dtype mapping is checked. CUDA library APIs that require cuComplex ABI pointers should
543 // cast explicitly at the library-call boundary.
544 template <typename T>
545 T *gpu_ptr_as() const {
546 cytnx_error_msg(
547 this->dtype() != Type_class::cy_typeid_gpu_v<std::remove_cv_t<T>>,
548 "[ERROR] Attempt to convert dtype %d (%s) to GPU pointer of type %s", this->dtype(),
549 Type_class::getname(this->dtype()).c_str(),
550 Type_class::getname(Type_class::cy_typeid_gpu_v<std::remove_cv_t<T>>).c_str());
551 return static_cast<T *>(this->_impl->_storage._impl->data());
552 }

1463 void append(const Tensor &rhs) {
1464 // Tensor in;
1465 if (!this->is_contiguous()) this->contiguous_();
1466
1467 // check Tensor in shape:
1468 cytnx_error_msg(rhs.shape().size() == 0 || this->shape().size() == 0,
1469 "[ERROR] try to append a null Tensor.%s", "\n");
1470 cytnx_error_msg(rhs.shape().size() != (this->shape().size() - 1),
1471 "[ERROR] try to append a Tensor with rank not match.%s", "\n");
1472 cytnx_uint64 Nelem = 1;
1473 for (unsigned int i = 0; i < rhs.shape().size(); i++) {
1474 cytnx_error_msg(rhs.shape()[i] != this->shape()[i + 1],
1475 "[ERROR] dimension mismatch @ rhs.rank: [%d] this: [%d] rhs: [%d]\n", i,
1476 this->shape()[i + 1], rhs.shape()[i]);
1477 Nelem *= rhs.shape()[i];
1478 }
1479
1480 // check type:
1481 Tensor in;
1482 if (rhs.dtype() != this->dtype()) {
1483 in = rhs.astype(this->dtype());
1484 if (!in.is_contiguous()) in.contiguous_();
1485 } else {
1486 if (!in.is_contiguous())
1487 in = rhs.contiguous();
1488 else
1489 in = rhs;
1490 }
1491 this->_impl->_shape[0] += 1;
1492 cytnx_uint64 oldsize = this->_impl->_storage.size();
1493 this->_impl->_storage.resize(oldsize + Nelem);
1494 memcpy(((char *)this->_impl->_storage.data()) +
1495 oldsize * Type.typeSize(this->dtype()) / sizeof(char),
1496 in._impl->_storage.data(), Type.typeSize(in.dtype()) * Nelem);
1497 }
