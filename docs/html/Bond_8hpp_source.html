72 void set_type(const bondType &new_bondType) {
73 if ((this->_type != BD_REG)) {
74 if (new_bondType == BD_REG) {
75 cytnx_error_msg(this->_qnums.size(),
76 "[ERROR] cannot change type to BD_REG for a symmetry bond.%s", "\n");
77 }
78 if (std::abs(int(this->_type)) != std::abs(int(new_bondType))) {
79 cytnx_error_msg(this->_qnums.size(),
80 "[ERROR] cannot exchange BDtype between BD_* <-> gBD_* .%s", "\n");
81 }
82 }
83
84 this->_type = new_bondType;
85 }
