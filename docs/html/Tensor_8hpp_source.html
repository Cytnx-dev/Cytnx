126  cytnx_uint64 RealRank,mtplyr;
127  std::vector<cytnx_uint64> c_shape(this->_shape.size());
128  std::vector<cytnx_uint64> c_loc(this->_shape.size());
129 
130  RealRank=0;
131  mtplyr = 1;

133  for(cytnx_int64 i=this->_shape.size()-1; i>=0; i--){
134  if(locator[i]>=this->_shape[i]){
135  cytnx_error_msg(true, "%s", "The dimension of rank that trying to access is exceed Tensor's dimension.");

197  if(!this->_contiguous){
198  this->contiguous_();
199  }
200  std::vector<cytnx_uint64> result_shape(new_shape.size());
201  cytnx_uint64 new_N = 1;

204  for(int i=0;i<new_shape.size();i++){
205  if(new_shape[i]<0){
206  if(new_shape[i]!=-1) cytnx_error_msg(new_shape[i]!=-1,"%s","[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
207  if(has_undetermine) cytnx_error_msg(new_shape[i]!=-1,"%s","[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
208  Udet_id = i;
209  has_undetermine = true;
210  }else{
211  new_N *= new_shape[i];
212  result_shape[i] = new_shape[i];
213  }
214  }
215 
216

218  cytnx_error_msg(new_N >= this->_storage.size(),"%s","[ERROR] new shape exceed the total number of elements.");
219  cytnx_error_msg(this->_storage.size()%new_N,"%s","[ERROR] unmatch size when reshape with undetermine dimension");
220  result_shape[Udet_id] = this->_storage.size()/new_N;
221  }else{
222  cytnx_error_msg(new_N != this->_storage.size(),"%s","[ERROR] new shape does not match the number of elements.");
223  }
