134  T& at(const std::vector<cytnx_uint64> &locator){
135  cytnx_error_msg(locator.size() != this->_shape.size(), "%s", "The input indexes rank is not match Tensor's rank.");
136 
137  cytnx_uint64 RealRank,mtplyr;
138  std::vector<cytnx_uint64> c_shape(this->_shape.size());
139  std::vector<cytnx_uint64> c_loc(this->_shape.size());
140 
141  RealRank=0;
142  mtplyr = 1;
143 
144  for(cytnx_int64 i=this->_shape.size()-1; i>=0; i--){
145  if(locator[i]>=this->_shape[i]){
146  cytnx_error_msg(true, "%s", "The dimension of rank that trying to access is exceed Tensor's dimension.");
147  }
148  c_shape[i] = this->_shape[this->_invmapper[i]];
149  c_loc[i] = locator[this->_invmapper[i]];
150  RealRank += mtplyr*c_loc[i];
151  mtplyr *= c_shape[i];
152  }
153  return this->_storage.at<T>(RealRank);

207  void reshape_(const std::vector<cytnx_int64> &new_shape){
208  if(!this->_contiguous){
209  this->contiguous_();
210  }
211  std::vector<cytnx_uint64> result_shape(new_shape.size());
212  cytnx_uint64 new_N = 1;
213  bool has_undetermine = false;
214  unsigned int Udet_id = 0;
215  for(int i=0;i<new_shape.size();i++){
216  if(new_shape[i]<0){
217  if(new_shape[i]!=-1) cytnx_error_msg(new_shape[i]!=-1,"%s","[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
218  if(has_undetermine) cytnx_error_msg(new_shape[i]!=-1,"%s","[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
219  Udet_id = i;
220  has_undetermine = true;
221  }else{
222  new_N *= new_shape[i];
223  result_shape[i] = new_shape[i];
224  }
225  }
226 
227 
228  if(has_undetermine){
229  cytnx_error_msg(new_N >= this->_storage.size(),"%s","[ERROR] new shape exceed the total number of elements.");
230  cytnx_error_msg(this->_storage.size()%new_N,"%s","[ERROR] unmatch size when reshape with undetermine dimension");
231  result_shape[Udet_id] = this->_storage.size()/new_N;
232  }else{
233  cytnx_error_msg(new_N != this->_storage.size(),"%s","[ERROR] new shape does not match the number of elements.");
234  }
235 
236  this->_shape = result_shape;
237  this->_mapper = utils_internal::range_cpu(result_shape.size());
238  this->_invmapper = this->_mapper;
