112 T &at(const std::vector<cytnx_uint64> &locator) const {
113 cytnx_error_msg(locator.size() != this->_shape.size(), "%s",
114 "The input index does not match Tensor's rank.");

116 cytnx_uint64 RealRank, mtplyr;
117 // std::vector<cytnx_uint64> c_shape(this->_shape.size());
118 // std::vector<cytnx_uint64> c_loc(this->_shape.size());
119 cytnx_uint64 c_shape, c_loc;
120
121 RealRank = 0;
122 mtplyr = 1;
123
124 for (cytnx_int64 i = this->_shape.size() - 1; i >= 0; i--) {
125 if (locator[i] >= this->_shape[i]) {
126 cytnx_error_msg(true, "%s", "Attempting to access out-of-bound index in Tensor.");
127 }
128 // c_shape[i] = this->_shape[this->_invmapper[i]];
129 // c_loc[i] = locator[this->_invmapper[i]];
130 c_shape = this->_shape[this->_invmapper[i]];
131 c_loc = locator[this->_invmapper[i]];
132 RealRank += mtplyr * c_loc;
133 mtplyr *= c_shape;
134 }
135 return this->_storage.at<T>(RealRank);

250 void reshape_(const std::vector<cytnx_int64> &new_shape) {
251 if (!this->_contiguous) {
252 this->contiguous_();
253 }
254 std::vector<cytnx_uint64> result_shape(new_shape.size());
255 cytnx_uint64 new_N = 1;
256 bool has_undetermine = false;
257 unsigned int Udet_id = 0;
258 for (int i = 0; i < new_shape.size(); i++) {
259 if (new_shape[i] < 0) {
260 if (new_shape[i] != -1)
261 cytnx_error_msg(
262 new_shape[i] != -1, "%s",
263 "[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
264 if (has_undetermine)
265 cytnx_error_msg(
266 new_shape[i] != -1, "%s",
267 "[ERROR] reshape can only have dimension > 0 and one undetermine rank specify as -1");
268 Udet_id = i;
269 has_undetermine = true;
270 } else {
271 new_N *= new_shape[i];
272 result_shape[i] = new_shape[i];
273 }
274 }
275
276 if (has_undetermine) {
277 cytnx_error_msg(new_N > this->_storage.size(), "%s",
278 "[ERROR] new shape exceed the total number of elements.");
279 cytnx_error_msg(this->_storage.size() % new_N, "%s",
280 "[ERROR] unmatch size when reshape with undetermine dimension");
281 result_shape[Udet_id] = this->_storage.size() / new_N;
282 } else {
283 cytnx_error_msg(new_N != this->_storage.size(), "%s",
284 "[ERROR] new shape does not match the number of elements.");
285 }
286
287 this->_shape = result_shape;
288 this->_mapper = vec_range(result_shape.size());
289 this->_invmapper = this->_mapper;
290 }

1357 if (!this->is_contiguous()) this->contiguous_();

1359 // check Tensor in shape:
1360 cytnx_error_msg(rhs.shape().size() == 0 || this->shape().size() == 0,
1361 "[ERROR] try to append a null Tensor.%s", "\n");
1362 cytnx_error_msg(rhs.shape().size() != (this->shape().size() - 1),
1363 "[ERROR] try to append a Tensor with rank not match.%s", "\n");
1364 cytnx_uint64 Nelem = 1;
1365 for (unsigned int i = 0; i < rhs.shape().size(); i++) {
1366 cytnx_error_msg(rhs.shape()[i] != this->shape()[i + 1],
1367 "[ERROR] dimension mismatch @ rhs.rank: [%d] this: [%d] rhs: [%d]\n", i,
1368 this->shape()[i + 1], rhs.shape()[i]);
1369 Nelem *= rhs.shape()[i];
1370 }
1371
1372 // check type:
1373 Tensor in;
1374 if (rhs.dtype() != this->dtype()) {
1375 in = rhs.astype(this->dtype());
1376 if (!in.is_contiguous()) in.contiguous_();
1377 } else {
1378 if (!in.is_contiguous())
1379 in = rhs.contiguous();
1380 else
1381 in = rhs;
